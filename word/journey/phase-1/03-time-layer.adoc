[[process]]
== Process: How We Built

=== 1. Define Base Temporal States

Three time relations from Genesis 1:1 "in the beginning":

[source,c]
----
typedef enum {
    TEMP_PAST    = TRIT_NEG,   // -1: What was - history, memory
    TEMP_PRESENT = TRIT_ZERO,  //  0: What is - now, current (anchor)
    TEMP_FUTURE  = TRIT_POS    // +1: What will be - anticipation
} temporal_t;
----

=== 2. Design Compound States

**The insight:** Consciousness doesn't just exist IN a time — it looks FROM one time AT another. This creates compound modes:

[source,c]
----
typedef struct {
    temporal_t lens;       // viewing FROM (past/present/future)
    temporal_t target;     // viewing AT (past/present/future)
    bool is_compound;      // true if using both lens and target
} temporal_state_t;
----

=== 4. Implement Creation Functions

[source,c]
----
temporal_state_t temporal_create_base(temporal_t t) {
    temporal_state_t state;
    state.lens = t;
    state.target = t;  // Same as lens for base states
    state.is_compound = false;
    return state;
}

temporal_state_t temporal_create_compound(temporal_t lens, temporal_t target) {
    temporal_state_t state;
    state.lens = lens;
    state.target = target;
    state.is_compound = true;
    return state;
}
----

=== 5. Implement Mode Names

[source,c]
----
const char* temporal_mode_name(temporal_state_t state) {
    if (!state.is_compound) {
        switch (state.lens) {
            case TEMP_PAST:    return "past";
            case TEMP_PRESENT: return "present";
            case TEMP_FUTURE:  return "future";
        }
    }

    // Compound modes - lens→target combinations
    if (state.lens == TEMP_PAST && state.target == TEMP_PRESENT)
        return "memory";
    if (state.lens == TEMP_PAST && state.target == TEMP_FUTURE)
        return "prophecy";
    if (state.lens == TEMP_PRESENT && state.target == TEMP_PAST)
        return "reflection";
    if (state.lens == TEMP_PRESENT && state.target == TEMP_FUTURE)
        return "planning";
    if (state.lens == TEMP_FUTURE && state.target == TEMP_PAST)
        return "hindsight";
    if (state.lens == TEMP_FUTURE && state.target == TEMP_PRESENT)
        return "preparation";

    return "present";  // Safe default
}
----
