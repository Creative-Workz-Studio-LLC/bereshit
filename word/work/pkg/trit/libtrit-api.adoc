[[balanced-ternary-arithmetic]]
=== Balanced Ternary Arithmetic

Working with single trits and basic operations.

[source,c]
----
#include "trit.h"

// Create trits representing Genesis 1:1 dimensions
trit_t heaven = trit_create(TRIT_POS);  // +1 = space
trit_t earth = trit_create(TRIT_NEG);   // -1 = matter

// Arithmetic: opposites cancel
trit_t sum = trit_add(heaven, earth);   // +1 + -1 = 0

// Multiplication follows sign rules
trit_t prod = trit_multiply(earth, earth); // -1 Ã— -1 = +1

// Negation reverses direction
trit_t flipped = trit_negate(heaven);   // +1 â†’ -1
----

'''

[[packing-for-storage]]
=== Packing for Storage

Efficient storage using pack/unpack operations.

[source,c]

// Check for Bible Rail spare states
if (trit5_is_spare(packed)) {
    // This is a WEB translation variant (243-255)
}

// Recover original trits
trit_t recovered[5];
trit5_unpack(packed, recovered);
// recovered == {-1, 0, +1, +1, -1}
